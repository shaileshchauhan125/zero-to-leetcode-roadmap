__init__ is a special method that runs when creating a new object. It sets up the initial state of the object.
3
Add Methods (Functions)
class Student:
    def __init__(self, name, age, major):
        self.name = name
        self.age = age
        self.major = major
        self.grades = []
    
    def add_grade(self, grade):
        """Add a grade to the student's record"""
        self.grades.append(grade)
    
    def get_average(self):
        """Calculate the student's average grade"""
        if not self.grades:
            return 0
        return sum(self.grades) / len(self.grades)
4
Add String Representation
class Student:
    def __init__(self, name, age, major):
        self.name = name
        self.age = age
        self.major = major
        self.grades = []
    
    def add_grade(self, grade):
        self.grades.append(grade)
    
    def get_average(self):
        if not self.grades:
            return 0
        return sum(self.grades) / len(self.grades)
    
    def __str__(self):
        """String representation of the student"""
        return f"Student: {self.name}, Major: {self.major}"
